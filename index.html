cloud Google Sheets & Drive Integration Setup

Follow these steps to connect your Trading Journal with Google Sheets and Google Drive:

  1. Create a Google Sheet:
    • Go to sheets.google.com
    • Create a new spreadsheet named "Trading Journal"
    • Copy the Spreadsheet ID from the URL (between /d/ and /edit)
  2. Enable Google Apps Script:
    • In your Google Sheet, go to Extensions → Apps Script
    • Delete any default code
    • Copy and paste the provided Apps Script code (see below)
    • Click "Deploy" → "New deployment" → "Web app"
    • Set "Execute as" to "Me" and "Who has access" to "Anyone"
    • Copy the deployment URL
  3. Configure the Application:
    • Paste your deployment URL in the field below
    • Click "Connect to Google Sheets"

Copy this code to your Google Apps Script editor:

// Trading Journal - Google Apps Script // This script handles data sync between the web app and Google Sheets const SHEET_NAME = 'Trades'; const SCREENSHOTS_FOLDER_NAME = 'Trading Journal Screenshots'; // Initialize spreadsheet function doGet(e) { return ContentService.createTextOutput(JSON.stringify({ status: 'success', message: 'Trading Journal API is ready' })).setMimeType(ContentService.MimeType.JSON); } // Handle POST requests function doPost(e) { try { const data = JSON.parse(e.postData.contents); const action = data.action; switch(action) { case 'addTrade': return addTrade(data.trade); case 'getAllTrades': return getAllTrades(); case 'updateTrade': return updateTrade(data.id, data.trade); case 'deleteTrade': return deleteTrade(data.id); case 'uploadScreenshot': return uploadScreenshot(data.screenshot, data.filename); default: return createResponse(false, 'Unknown action'); } } catch(error) { return createResponse(false, error.toString()); } } // Get or create sheet function getSheet() { const ss = SpreadsheetApp.getActiveSpreadsheet(); let sheet = ss.getSheetByName(SHEET_NAME); if (!sheet) { sheet = ss.insertSheet(SHEET_NAME); sheet.appendRow(['ID', 'Date', 'Time', 'Pair', 'Session', 'Type', 'Result', 'Entry Price', 'Exit Price', 'Notes', 'Screenshot URL', 'Created At']); } return sheet; } // Add new trade function addTrade(trade) { const sheet = getSheet(); const id = new Date().getTime().toString(); const createdAt = new Date().toISOString(); sheet.appendRow([ id, trade.date, trade.time, trade.pair, trade.session, trade.type, trade.result, trade.entryPrice || '', trade.exitPrice || '', trade.notes || '', trade.screenshotUrl || '', createdAt ]); return createResponse(true, 'Trade added successfully', { id: id, createdAt: createdAt }); } // Get all trades function getAllTrades() { const sheet = getSheet(); const data = sheet.getDataRange().getValues(); const trades = []; for (let i = 1; i < data.length; i++) { trades.push({ id: data[i][0], date: data[i][1], time: data[i][2], pair: data[i][3], session: data[i][4], type: data[i][5], result: data[i][6], entryPrice: data[i][7], exitPrice: data[i][8], notes: data[i][9], screenshotUrl: data[i][10], createdAt: data[i][11] }); } return createResponse(true, 'Trades retrieved successfully', trades); } // Update trade function updateTrade(id, trade) { const sheet = getSheet(); const data = sheet.getDataRange().getValues(); for (let i = 1; i < data.length; i++) { if (data[i][0] == id) { sheet.getRange(i + 1, 2, 1, 10).setValues([[ trade.date, trade.time, trade.pair, trade.session, trade.type, trade.result, trade.entryPrice || '', trade.exitPrice || '', trade.notes || '', trade.screenshotUrl || '' ]]); return createResponse(true, 'Trade updated successfully'); } } return createResponse(false, 'Trade not found'); } // Delete trade function deleteTrade(id) { const sheet = getSheet(); const data = sheet.getDataRange().getValues(); for (let i = 1; i < data.length; i++) { if (data[i][0] == id) { sheet.deleteRow(i + 1); return createResponse(true, 'Trade deleted successfully'); } } return createResponse(false, 'Trade not found'); } // Upload screenshot to Google Drive function uploadScreenshot(base64Data, filename) { try { const blob = Utilities.newBlob( Utilities.base64Decode(base64Data.split(',')[1]), 'image/png', filename ); const folders = DriveApp.getFoldersByName(SCREENSHOTS_FOLDER_NAME); let folder; if (folders.hasNext()) { folder = folders.next(); } else { folder = DriveApp.createFolder(SCREENSHOTS_FOLDER_NAME); } const file = folder.createFile(blob); file.setSharing(DriveApp.Access.ANYONE_WITH_LINK, DriveApp.Permission.VIEW); return createResponse(true, 'Screenshot uploaded successfully', { url: file.getUrl(), id: file.getId() }); } catch(error) { return createResponse(false, 'Screenshot upload failed: ' + error.toString()); } } // Create response function createResponse(success, message, data = null) { const response = { success: success, message: message, data: data }; return ContentService .createTextOutput(JSON.stringify(response)) .setMimeType(ContentService.MimeType.JSON); }
