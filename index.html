// Multi-User Trading Journal - Google Apps Script // This script handles multi-user data separation const SHEET_NAME = 'UsersData'; const SCREENSHOTS_FOLDER_NAME = 'Trading Journal Screenshots'; const ADMIN_PASSWORD = 'admin123'; // Change this password // Initialize spreadsheet function doGet(e) { return ContentService.createTextOutput(JSON.stringify({ status: 'success', message: 'Multi-User Trading Journal API is ready' })).setMimeType(ContentService.MimeType.JSON); } // Handle POST requests function doPost(e) { try { const data = JSON.parse(e.postData.contents); const action = data.action; switch(action) { case 'login': return loginUser(data.email, data.password, data.name); case 'addTrade': return addTrade(data.userId, data.trade); case 'getUserTrades': return getUserTrades(data.userId); case 'updateTrade': return updateTrade(data.userId, data.tradeId, data.trade); case 'deleteTrade': return deleteTrade(data.userId, data.tradeId); case 'deleteAllUserTrades': return deleteAllUserTrades(data.userId); case 'uploadScreenshot': return uploadScreenshot(data.userId, data.screenshot, data.filename); case 'getAllUsers': return getAllUsers(); case 'deleteAllUsersData': return deleteAllUsersData(data.adminPassword); case 'getAllUsersTrades': return getAllUsersTrades(data.adminPassword); case 'getUserStats': return getUserStats(data.userId); default: return createResponse(false, 'Unknown action'); } } catch(error) { return createResponse(false, error.toString()); } } // Get or create main sheet function getSheet() { const ss = SpreadsheetApp.getActiveSpreadsheet(); let sheet = ss.getSheetByName(SHEET_NAME); if (!sheet) { sheet = ss.insertSheet(SHEET_NAME); sheet.appendRow(['User ID', 'Email', 'Name', 'Password Hash', 'Created At', 'Last Login']); } return sheet; } // Get or create user trades sheet function getUserTradesSheet(userId) { const ss = SpreadsheetApp.getActiveSpreadsheet(); const sheetName = `Trades_${userId}`; let sheet = ss.getSheetByName(sheetName); if (!sheet) { sheet = ss.insertSheet(sheetName); sheet.appendRow(['Trade ID', 'Date', 'Time', 'Pair', 'Session', 'Type', 'Result', 'Entry Price', 'Exit Price', 'Risk Amount', 'Reward Amount', 'Profit/Loss', 'Notes', 'Screenshot URL', 'Created At']); } return sheet; } // User authentication function loginUser(email, password, name) { const sheet = getSheet(); const data = sheet.getDataRange().getValues(); // Find existing user for (let i = 1; i < data.length; i++) { if (data[i][1] === email) { // Verify password const storedHash = data[i][3]; const inputHash = Utilities.computeDigest(Utilities.DigestAlgorithm.MD5, password).toString(); if (storedHash === inputHash) { // Update last login sheet.getRange(i + 1, 6).setValue(new Date().toISOString()); return createResponse(true, 'Login successful', { userId: data[i][0], email: data[i][1], name: data[i][2], createdAt: data[i][4], lastLogin: data[i][5] }); } else { return createResponse(false, 'Invalid password'); } } } // Create new user const userId = Utilities.getUuid(); const passwordHash = Utilities.computeDigest(Utilities.DigestAlgorithm.MD5, password).toString(); const now = new Date().toISOString(); sheet.appendRow([userId, email, name, passwordHash, now, now]); // Create user trades sheet getUserTradesSheet(userId); return createResponse(true, 'User registered successfully', { userId: userId, email: email, name: name, createdAt: now, lastLogin: now }); } // Add trade for user function addTrade(userId, trade) { const sheet = getUserTradesSheet(userId); const tradeId = Utilities.getUuid(); const createdAt = new Date().toISOString(); // Calculate profit/loss let profitLoss = 0; if (trade.result === 'Win' && trade.rewardAmount) { profitLoss = parseFloat(trade.rewardAmount); } else if (trade.result === 'Loss' && trade.riskAmount) { profitLoss = -parseFloat(trade.riskAmount); } sheet.appendRow([ tradeId, trade.date, trade.time, trade.pair, trade.session, trade.type, trade.result, trade.entryPrice || '', trade.exitPrice || '', trade.riskAmount || '', trade.rewardAmount || '', profitLoss, trade.notes || '', trade.screenshotUrl || '', createdAt ]); return createResponse(true, 'Trade added successfully', { tradeId: tradeId, createdAt: createdAt, profitLoss: profitLoss }); } // Get user's trades function getUserTrades(userId) { const sheet = getUserTradesSheet(userId); const data = sheet.getDataRange().getValues(); const trades = []; for (let i = 1; i < data.length; i++) { trades.push({ id: data[i][0], date: data[i][1], time: data[i][2], pair: data[i][3], session: data[i][4], type: data[i][5], result: data[i][6], entryPrice: data[i][7], exitPrice: data[i][8], riskAmount: data[i][9], rewardAmount: data[i][10], profitLoss: parseFloat(data[i][11]) || 0, notes: data[i][12], screenshotUrl: data[i][13], createdAt: data[i][14] }); } return createResponse(true, 'Trades retrieved successfully', trades); } // Update user's trade function updateTrade(userId, tradeId, trade) { const sheet = getUserTradesSheet(userId); const data = sheet.getDataRange().getValues(); for (let i = 1; i < data.length; i++) { if (data[i][0] == tradeId) { // Calculate profit/loss let profitLoss = 0; if (trade.result === 'Win' && trade.rewardAmount) { profitLoss = parseFloat(trade.rewardAmount); } else if (trade.result === 'Loss' && trade.riskAmount) { profitLoss = -parseFloat(trade.riskAmount); } sheet.getRange(i + 1, 2, 1, 13).setValues([[ trade.date, trade.time, trade.pair, trade.session, trade.type, trade.result, trade.entryPrice || '', trade.exitPrice || '', trade.riskAmount || '', trade.rewardAmount || '', profitLoss, trade.notes || '', trade.screenshotUrl || '' ]]); return createResponse(true, 'Trade updated successfully'); } } return createResponse(false, 'Trade not found'); } // Delete user's trade function deleteTrade(userId, tradeId) { const sheet = getUserTradesSheet(userId); const data = sheet.getDataRange().getValues(); for (let i = 1; i < data.length; i++) { if (data[i][0] == tradeId) { sheet.deleteRow(i + 1); // Delete screenshot if exists const screenshotUrl = data[i][13]; if (screenshotUrl) { try { const fileId = screenshotUrl.split('/d/')[1]?.split('/')[0]; if (fileId) { DriveApp.getFileById(fileId).setTrashed(true); } } catch(e) { console.log('Failed to delete screenshot:', e); } } return createResponse(true, 'Trade deleted successfully'); } } return createResponse(false, 'Trade not found'); } // Delete all user's trades function deleteAllUserTrades(userId) { const sheet = getUserTradesSheet(userId); // Delete all rows except header if (sheet.getLastRow() > 1) { sheet.deleteRows(2, sheet.getLastRow() - 1); } // Delete user's screenshots folder try { const userFolderName = `User_${userId}_Screenshots`; const folders = DriveApp.getFoldersByName(userFolderName); if (folders.hasNext()) { const folder = folders.next(); const files = folder.getFiles(); while (files.hasNext()) { files.next().setTrashed(true); } folder.setTrashed(true); } } catch(e) { console.log('Failed to delete user screenshots folder:', e); } return createResponse(true, 'All trades deleted successfully'); } // Upload screenshot to user's folder function uploadScreenshot(userId, base64Data, filename) { try { const blob = Utilities.newBlob( Utilities.base64Decode(base64Data.split(',')[1]), 'image/png', filename ); // Create user-specific folder const userFolderName = `User_${userId}_Screenshots`; let folder; const folders = DriveApp.getFoldersByName(userFolderName); if (folders.hasNext()) { folder = folders.next(); } else { folder = DriveApp.createFolder(userFolderName); } const file = folder.createFile(blob); file.setSharing(DriveApp.Access.ANYONE_WITH_LINK, DriveApp.Permission.VIEW); return createResponse(true, 'Screenshot uploaded successfully', { url: file.getUrl(), id: file.getId() }); } catch(error) { return createResponse(false, 'Screenshot upload failed: ' + error.toString()); } } // Get all users (admin only) function getAllUsers() { const sheet = getSheet(); const data = sheet.getDataRange().getValues(); const users = []; for (let i = 1; i < data.length; i++) { // Get user's trade count const userId = data[i][0]; const tradesSheet = getUserTradesSheet(userId); const tradeCount = Math.max(0, tradesSheet.getLastRow() - 1); users.push({ userId: userId, email: data[i][1], name: data[i][2], createdAt: data[i][4], lastLogin: data[i][5], tradeCount: tradeCount }); } return createResponse(true, 'Users retrieved successfully', users); } // Get user statistics function getUserStats(userId) { const trades = getUserTradesSheet(userId).getDataRange().getValues(); let totalTrades = Math.max(0, trades.length - 1); let wins = 0; let losses = 0; let totalProfitLoss = 0; for (let i = 1; i < trades.length; i++) { if (trades[i][6] === 'Win') wins++; if (trades[i][6] === 'Loss') losses++; totalProfitLoss += parseFloat(trades[i][11]) || 0; } const winRate = totalTrades > 0 ? (wins / totalTrades * 100).toFixed(1) : 0; return createResponse(true, 'User stats retrieved', { totalTrades: totalTrades, wins: wins, losses: losses, winRate: winRate, totalProfitLoss: totalProfitLoss }); } // Delete all users data (admin only) function deleteAllUsersData(adminPassword) { if (adminPassword !== ADMIN_PASSWORD) { return createResponse(false, 'Invalid admin password'); } const ss = SpreadsheetApp.getActiveSpreadsheet(); // Delete all sheets except UsersData const sheets = ss.getSheets(); sheets.forEach(sheet => { if (sheet.getName() !== SHEET_NAME) { ss.deleteSheet(sheet); } }); // Clear UsersData except header const userSheet = ss.getSheetByName(SHEET_NAME); if (userSheet.getLastRow() > 1) { userSheet.deleteRows(2, userSheet.getLastRow() - 1); } // Delete all screenshots folders const folders = DriveApp.getFolders(); while (folders.hasNext()) { const folder = folders.next(); if (folder.getName().includes('_Screenshots')) { const files = folder.getFiles(); while (files.hasNext()) { files.next().setTrashed(true); } folder.setTrashed(true); } } return createResponse(true, 'All users data deleted successfully'); } // Get all users trades (admin only) function getAllUsersTrades(adminPassword) { if (adminPassword !== ADMIN_PASSWORD) { return createResponse(false, 'Invalid admin password'); } const allTrades = []; const ss = SpreadsheetApp.getActiveSpreadsheet(); const sheets = ss.getSheets(); for (let sheet of sheets) { const sheetName = sheet.getName(); if (sheetName.startsWith('Trades_')) { const userId = sheetName.replace('Trades_', ''); const data = sheet.getDataRange().getValues(); for (let i = 1; i < data.length; i++) { allTrades.push({ userId: userId, tradeId: data[i][0], date: data[i][1], time: data[i][2], pair: data[i][3], session: data[i][4], type: data[i][5], result: data[i][6], profitLoss: parseFloat(data[i][11]) || 0, createdAt: data[i][14] }); } } } return createResponse(true, 'All trades retrieved', allTrades); } // Create response function createResponse(success, message, data = null) { const response = { success: success, message: message, data: data }; return ContentService .createTextOutput(JSON.stringify(response)) .setMimeType(ContentService.MimeType.JSON); }
